CREATE DATABASE BibliotecaDB;
GO
USE BibliotecaDB;
GO

CREATE TABLE dbo.Libros (
    LibroId INT IDENTITY(1,1) PRIMARY KEY,
    Titulo NVARCHAR(200) NOT NULL,
    Autor NVARCHAR(150) NOT NULL,
    AnioPublicacion INT NULL,
    Categoria NVARCHAR(100) NOT NULL,
    Disponible BIT NOT NULL DEFAULT(1)
);
GO

CREATE TABLE dbo.Solicitudes (
    SolicitudId INT IDENTITY(1,1) PRIMARY KEY,
    Titulo NVARCHAR(200) NOT NULL,
    Autor NVARCHAR(150) NULL,
    AnioPublicacion INT NULL,
    Categoria NVARCHAR(100) NULL,
    NombreSolicitante NVARCHAR(120) NULL,
    CorreoSolicitante NVARCHAR(120) NULL,
    Comentario NVARCHAR(500) NULL,
    FechaSolicitud DATETIME2 NOT NULL DEFAULT(SYSDATETIME())
);
GO
